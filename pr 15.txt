declare
  cursor c1 is select * from n_rollcall;
  v_row c1%rowtype;
begin
  open c1;
  loop
    fetch c1 into v_row;
    exit when c1%notfound;
    begin
      insert into o_rollcall values (v_row.roll_no, v_row.name);
    exception
      when dup_val_on_index then
        null;
    end;
  end loop;
  close c1;
end;
